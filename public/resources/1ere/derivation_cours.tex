\section{Nombre dérivé}
$f'(a) = \lim_{h \to 0} \frac{f(a+h) - f(a)}{h}$.

\section{Dérivées usuelles}
\begin{itemize}
    \item $(x^n)' = nx^{n-1}$
    \item $(\frac{1}{x})' = -\frac{1}{x^2}$
    \item $(\sqrt{x})' = \frac{1}{2\sqrt{x}}$
\end{itemize}
