\section*{Partie 2 : Exercices}

\subsection*{Exercice 1}
\begin{enumerate}
    \item $u_1 = \dfrac12\times 10 + 2 = 5+2=7$ ; $u_2 = \dfrac12\times 7 + 2 = 3,5+2 = 5,5$.
    \item $u_1-u_0 = -3$ ; $u_2-u_1 = -1,5$ : différences non constantes $\Rightarrow$ pas arithmétique. $\dfrac{u_1}{u_0}=0,7$ ; $\dfrac{u_2}{u_1}\approx 0,7857$ : rapports non constants $\Rightarrow$ pas géométrique.
    \item \begin{enumerate}
        \item $v_{n+1} = u_{n+1}-4 = \left(\dfrac12 u_n+2\right)-4 = \dfrac12 u_n -2 = \dfrac12 (u_n-4) = \dfrac12 v_n$. Donc $(v_n)$ est géométrique de raison $\dfrac12$ et $v_0 = u_0-4 = 6$.
        \item $v_n = v_0 \times \left(\dfrac12\right)^n = 6\times \dfrac{1}{2^n}$. Alors $u_n = v_n+4 = 4 + \dfrac{6}{2^n}$.
    \end{enumerate}
    \item $u_{n+1}-u_n = \left(4+\dfrac{6}{2^{n+1}}\right) - \left(4+\dfrac{6}{2^n}\right) = \dfrac{6}{2^{n+1}} - \dfrac{12}{2^{n+1}} = -\dfrac{6}{2^{n+1}} <0$. Donc $(u_n)$ strictement décroissante.
   
\end{enumerate}
